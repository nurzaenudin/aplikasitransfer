CREATE TABLE `latihan_user`.`s_user`(  
  `id` VARCHAR(36) NOT NULL,
  `username` VARCHAR(250) NOT NULL,
  `active` BOOLEAN NOT NULL DEFAULT TRUE,
  PRIMARY KEY (`id`)
);

CREATE TABLE `latihan_user`.`s_user_password`(  
  `id` VARCHAR(36) NOT NULL,
  `user_id` VARCHAR(36) NOT NULL,
  `password` VARCHAR(15) NOT NULL,
  PRIMARY KEY (`id`),
  FOREIGN KEY (`user_id`) REFERENCES `latihan_user`.`s_user`(`id`) ON UPDATE CASCADE ON DELETE CASCADE
);
